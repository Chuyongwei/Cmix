#define N 5

typedef struct PolynNode{  
    int coef; 
    int expn;
    struct PolynNode *next;  
}PolynNode;

void CreatePolyn(PolynList &L,int n){  
    int i;  
    PolynList p,q;  
    L=(PolynList)malloc(sizeof(PolynNode));   
    L->next=NULL;  
    q=L;  
    printf("成对输入%d个数据\n",n);  
    for(i=1;i<=n;i++){  
        p=(PolynList)malloc(sizeof(PolynNode));  
        scanf("%d%d",&p->coef,&p->expn);      
        q->next=p;  
        q=q->next;  
    }  
    p->next=NULL;  
}
